local function fetchCorrectKey()
    local success, response = pcall(function()
        return HttpService:GetAsync(keyUrl)
    end)
    
    if not success then
        warn("Failed to fetch key:", response)
        return nil
    end
    
    -- For JSON format
    local success2, keyData = pcall(function()
        return HttpService:JSONDecode(response)
    end)
    
    if success2 and keyData.key then
        return keyData.key:gsub("%s+", "")
    end
    
    -- Fallback to plain text
    return response:gsub("%s+", "")
end
